## 选择排序

=== "C++"

    ```cpp
    vector<int> selectionSort(vector<int> arr) {
        for (int i = 0; i < arr.size(); i++) {
            int minIndex = i;
            for (int j = i + 1; j < arr.size(); j++) {
                if (arr[minIndex] > arr[j]) {
                    minIndex = j;
                }
            }
            if (minIndex != i) {
                swap(arr[i], arr[minIndex]);
            }
        }
        return arr;
    }
    ```
